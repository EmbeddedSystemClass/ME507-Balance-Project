81 /** @brief Applies PI control to output a duty cycle for each motor.
82  * @details Uses the error signal calculated from the setpoints for x and y to the
83  * current accelerations sensed in the x and y directions to calculate the
84  * appropriate actuation signal. The control gains convert the error in mG to
85  * voltage.
86  */
87 
88  void Balance::control ()
89  {
90  // Integral sum of the error, delta T = .005s from the rate at which the controller is run
91  esum_x += (set_x - x_accel) * .005;
92  // PI control
93  int16_t signalVal = (set_x - x_accel)*kp + esum_x*ki;
94  motor_A_actuation_signal->put(signalVal);
95 
96  // Integral sum of the error, delta T = .005s from the rate at which the controller is run
97  esum_y += (set_y - y_accel) * .01;
98  // PI control
99  signalVal = (set_y - y_accel)*kp + esum_y*ki;
100  motor_B_actuation_signal->put(signalVal);
101  }
